
### var make 和 :=
```go
    
    var x []int //正确可以append，之后可以读写
    var x map[int]int  //错误读不panic， 写会发生panic
	var x chan int //正确，但是是nil不可直接使用
	var x struct{}{} //正确
    
    y := []int{}//正确
    y := map[int]int{}//正确
    
    z := make([]int, 0)//正确
    z := make(map[int]int)//正确
    z := make(chan int)//正确，可直接使用
	
	var x struct{
		time <-chan
    }{}//可以直接使用
	
```
